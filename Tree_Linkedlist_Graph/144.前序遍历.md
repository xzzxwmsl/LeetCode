class Solution {
    List<Integer> re;
    public List<Integer> preorderTraversal(TreeNode root) {
        re =  new LinkedList<>();
        fun(root);
        return re;
    }

    public void fun(TreeNode root){
        if(root == null) return;
        re.add(root.val);
        fun(root.left);
        fun(root.right);
    }
}
