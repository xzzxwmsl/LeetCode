class Solution {
    public int romanToInt(String s) {
        int result = 0;

        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            int plus=0;
            // if( c=='I' && (s.charAt(i+1)=='V' || s.charAt(i+1)=='X' ) ){

            // }
            if(c=='X'||c=='V'){
                plus = getNum(c);
                if(i!=0 && s.charAt(i-1)=='I') plus-=2;
            }else if(c=='L' || c=='C'){
                plus = getNum(c);
                if(i!=0 && s.charAt(i-1)=='X') plus-=20;
            }else if(c=='D' || c=='M'){
                plus = getNum(c);
                if(i != 0 && s.charAt(i-1)=='C') plus-=200;
            }else{
                plus = getNum (c);
            }

            result += plus;
        }

        return result;

    }

     public static int getNum(char c) {
        int result = 0;
        switch (c) {
            case 'I':
                result = 1;
                break;
            case 'V':
                result = 5;
                break;
            case 'X':
                result = 10;
                break;
            case 'L':
                result = 50;
                break;
            case 'C':
                result = 100;
                break;
            case 'D':
                result = 500;
                break;
            case 'M':
                result = 1000;
                break;
        }
        return result;
    }
}
